# WebRTC P2P Remote Control Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace Kite's direct HTTP/WebSocket connection with WebRTC DataChannel for NAT-traversing P2P remote control.

**Architecture:** A public Go signaling server relays SDP/ICE messages between the Kite daemon (Zig + libdatachannel) and the phone browser. Authentication tokens travel only through the DTLS-encrypted DataChannel, never through the signaling server. The daemon opens no ports; it connects outbound to the signaling server.

**Tech Stack:** Zig 0.15.2 + libdatachannel (C), Go 1.22+ (signaling server), Svelte 5 + TypeScript (frontend), WebRTC DataChannel API (browser)

**Spec:** `docs/superpowers/specs/2026-04-05-webrtc-p2p-design.md`

```go
package main

import (
	"errors"
	"sync"
	"time"
)

var (
	ErrRoomExists    = errors.New("room already exists")
	ErrRoomNotFound  = errors.New("room not found")
	ErrRoomLocked    = errors.New("room is locked")
	ErrRateLimited   = errors.New("rate limited")
)

type Sender interface {
	Send(data []byte) error
}

type Room struct {
	PairingCode string
	Daemon      Sender
	Browser     Sender
	Locked      bool
	LastActive  time.Time
}

type RoomManager struct {
	mu      sync.Mutex
	rooms   map[string]*Room
	roomTTL time.Duration

	// Rate limiting: IP -> timestamps of join attempts
	joinAttempts map[string][]time.Time
	rateLimit    int           // max attempts per window
	rateWindow   time.Duration
}

func NewRoomManager() *RoomManager {
	return &RoomManager{
		rooms:        make(map[string]*Room),
		roomTTL:      10 * time.Minute,
		joinAttempts: make(map[string][]time.Time),
		rateLimit:    10,
		rateWindow:   time.Minute,
	}
}

func (rm *RoomManager) Register(pairingCode string, daemon Sender) error {
	rm.mu.Lock()
	defer rm.mu.Unlock()

	if _, exists := rm.rooms[pairingCode]; exists {
		return ErrRoomExists
	}

	rm.rooms[pairingCode] = &Room{
		PairingCode: pairingCode,
		Daemon:      daemon,
		LastActive:  time.Now(),
	}
	return nil
}

func (rm *RoomManager) Join(pairingCode string, browser Sender) error {
	rm.mu.Lock()
	defer rm.mu.Unlock()

	room, exists := rm.rooms[pairingCode]
	if !exists {
		return ErrRoomNotFound
	}
	if room.Locked {
		return ErrRoomLocked
	}

	room.Browser = browser
	room.Locked = true
	room.LastActive = time.Now()

	// Notify daemon that browser has joined
	_ = room.Daemon.Send([]byte(`{"type":"peer_joined"}`))
	return nil
}

func (rm *RoomManager) CheckRateLimit(ip string) error {
	rm.mu.Lock()
	defer rm.mu.Unlock()

	now := time.Now()
	cutoff := now.Add(-rm.rateWindow)

	attempts := rm.joinAttempts[ip]
	// Filter old attempts
	valid := attempts[:0]
	for _, t := range attempts {
		if t.After(cutoff) {
			valid = append(valid, t)
		}
	}
	rm.joinAttempts[ip] = valid

	if len(valid) >= rm.rateLimit {
		return ErrRateLimited
	}

	rm.joinAttempts[ip] = append(valid, now)
	return nil
}

func (rm *RoomManager) RelayFromBrowser(pairingCode string, data []byte) {
	rm.mu.Lock()
	room, exists := rm.rooms[pairingCode]
	if exists {
		room.LastActive = time.Now()
	}
	rm.mu.Unlock()

	if exists && room.Daemon != nil {
		_ = room.Daemon.Send(data)
	}
}

func (rm *RoomManager) RelayFromDaemon(pairingCode string, data []byte) {
	rm.mu.Lock()
	room, exists := rm.rooms[pairingCode]
	if exists {
		room.LastActive = time.Now()
	}
	rm.mu.Unlock()

	if exists && room.Browser != nil {
		_ = room.Browser.Send(data)
	}
}

func (rm *RoomManager) DaemonDisconnected(pairingCode string) {
	rm.mu.Lock()
	room, exists := rm.rooms[pairingCode]
	if exists {
		delete(rm.rooms, pairingCode)
	}
	rm.mu.Unlock()

	if exists && room.Browser != nil {
		_ = room.Browser.Send([]byte(`{"type":"peer_left"}`))
	}
}

func (rm *RoomManager) BrowserDisconnected(pairingCode string) {
	rm.mu.Lock()
	defer rm.mu.Unlock()

	room, exists := rm.rooms[pairingCode]
	if !exists {
		return
	}

	room.Browser = nil
	room.Locked = false
	room.LastActive = time.Now()

	if room.Daemon != nil {
		_ = room.Daemon.Send([]byte(`{"type":"peer_left"}`))
	}
}

func (rm *RoomManager) CleanupStale() {
	rm.mu.Lock()
	defer rm.mu.Unlock()

	now := time.Now()
	for code, room := range rm.rooms {
		if now.Sub(room.LastActive) > rm.roomTTL {
			delete(rm.rooms, code)
		}
	}
}
```

```go
package main

import (
	"embed"
	"encoding/json"
	"flag"
	"fmt"
	"io/fs"
	"log"
	"net/http"
	"sync"
	"time"

	"github.com/gorilla/websocket"
)

//go:embed static
var staticFiles embed.FS

var upgrader = websocket.Upgrader{
	CheckOrigin: func(r *http.Request) bool { return true },
}

// wsConn wraps gorilla/websocket to implement Sender interface
type wsConn struct {
	conn *websocket.Conn
	mu   sync.Mutex
}

func (w *wsConn) Send(data []byte) error {
	w.mu.Lock()
	defer w.mu.Unlock()
	return w.conn.WriteMessage(websocket.TextMessage, data)
}

type signalMsg struct {
	Type        string `json:"type"`
	PairingCode string `json:"pairing_code,omitempty"`
}

type Handler struct {
	rm       *RoomManager
	staticFS http.Handler
}

func NewHandler(rm *RoomManager, staticDir string) http.Handler {
	h := &Handler{rm: rm}

	if staticDir != "" {
		h.staticFS = http.FileServer(http.Dir(staticDir))
	} else {
		sub, err := fs.Sub(staticFiles, "static")
		if err != nil {
			log.Fatal(err)
		}
		h.staticFS = http.FileServer(http.FS(sub))
	}

	mux := http.NewServeMux()
	mux.HandleFunc("/ws", h.handleWebSocket)
	mux.Handle("/", h.staticFS)
	return mux
}

func (h *Handler) handleWebSocket(w http.ResponseWriter, r *http.Request) {
	conn, err := upgrader.Upgrade(w, r, nil)
	if err != nil {
		return
	}
	defer conn.Close()

	ws := &wsConn{conn: conn}

	// Read first message to determine role
	conn.SetReadDeadline(time.Now().Add(30 * time.Second))
	_, raw, err := conn.ReadMessage()
	if err != nil {
		return
	}
	conn.SetReadDeadline(time.Time{}) // clear deadline

	var msg signalMsg
	if err := json.Unmarshal(raw, &msg); err != nil {
		ws.Send([]byte(`{"type":"error","message":"invalid json"}`))
		return
	}

	switch msg.Type {
	case "register":
		h.handleDaemon(ws, conn, msg.PairingCode)
	case "join":
		clientIP := r.Header.Get("X-Forwarded-For")
		if clientIP == "" {
			clientIP = r.RemoteAddr
		}
		if err := h.rm.CheckRateLimit(clientIP); err != nil {
			ws.Send([]byte(`{"type":"error","message":"rate limited"}`))
			return
		}
		h.handleBrowser(ws, conn, msg.PairingCode)
	default:
		ws.Send([]byte(`{"type":"error","message":"first message must be register or join"}`))
	}
}

func (h *Handler) handleDaemon(ws *wsConn, conn *websocket.Conn, pairingCode string) {
	if err := h.rm.Register(pairingCode, ws); err != nil {
		ws.Send([]byte(fmt.Sprintf(`{"type":"error","message":"%s"}`, err.Error())))
		return
	}
	defer h.rm.DaemonDisconnected(pairingCode)

	// Set up ping/pong keepalive
	conn.SetPongHandler(func(string) error {
		conn.SetReadDeadline(time.Now().Add(60 * time.Second))
		return nil
	})

	// Ping ticker
	go func() {
		ticker := time.NewTicker(30 * time.Second)
		defer ticker.Stop()
		for range ticker.C {
			ws.mu.Lock()
			err := conn.WriteMessage(websocket.PingMessage, nil)
			ws.mu.Unlock()
			if err != nil {
				return
			}
		}
	}()

	// Relay loop
	for {
		conn.SetReadDeadline(time.Now().Add(60 * time.Second))
		_, raw, err := conn.ReadMessage()
		if err != nil {
			return
		}
		h.rm.RelayFromDaemon(pairingCode, raw)
	}
}

func (h *Handler) handleBrowser(ws *wsConn, conn *websocket.Conn, pairingCode string) {
	if err := h.rm.Join(pairingCode, ws); err != nil {
		ws.Send([]byte(fmt.Sprintf(`{"type":"error","message":"%s"}`, err.Error())))
		return
	}
	defer h.rm.BrowserDisconnected(pairingCode)

	conn.SetPongHandler(func(string) error {
		conn.SetReadDeadline(time.Now().Add(60 * time.Second))
		return nil
	})

	go func() {
		ticker := time.NewTicker(30 * time.Second)
		defer ticker.Stop()
		for range ticker.C {
			ws.mu.Lock()
			err := conn.WriteMessage(websocket.PingMessage, nil)
			ws.mu.Unlock()
			if err != nil {
				return
			}
		}
	}()

	for {
		conn.SetReadDeadline(time.Now().Add(60 * time.Second))
		_, raw, err := conn.ReadMessage()
		if err != nil {
			return
		}
		h.rm.RelayFromBrowser(pairingCode, raw)
	}
}

func main() {
	port := flag.Int("port", 8080, "listen port")
	staticDir := flag.String("static", "", "static files directory (default: embedded)")
	flag.Parse()

	rm := NewRoomManager()

	// Background cleanup every minute
	go func() {
		for {
			time.Sleep(time.Minute)
			rm.CleanupStale()
		}
	}()

	handler := NewHandler(rm, *staticDir)
	addr := fmt.Sprintf(":%d", *port)
	log.Printf("kite-signal listening on %s", addr)
	log.Fatal(http.ListenAndServe(addr, handler))
}
```

- [ ] **Step 1: Write message_queue.zig with inline tests**

`src/message_queue.zig`:
```zig
const std = @import("std");

/// Thread-safe FIFO queue for passing messages between threads.
/// Producers (libdatachannel callbacks) push; the main loop pops.
pub const MessageQueue = struct {
    items: std.ArrayList([]u8),
    mutex: std.Thread.Mutex = .{},
    allocator: std.mem.Allocator,

    pub fn init(allocator: std.mem.Allocator) MessageQueue {
        return .{
            .items = std.ArrayList([]u8).empty,
            .allocator = allocator,
        };
    }

    pub fn deinit(self: *MessageQueue) void {
        self.mutex.lock();
        defer self.mutex.unlock();
        for (self.items.items) |item| {
            self.allocator.free(item);
        }
        self.items.deinit(self.allocator);
    }

    /// Push a message. Caller's data is duped; safe to call from any thread.
    pub fn push(self: *MessageQueue, data: []const u8) !void {
        const copy = try self.allocator.dupe(u8, data);
        self.mutex.lock();
        defer self.mutex.unlock();
        try self.items.append(self.allocator, copy);
    }

    /// Pop all pending messages. Caller owns the returned slice and each item.
    pub fn drain(self: *MessageQueue) ![][]u8 {
        self.mutex.lock();
        defer self.mutex.unlock();
        if (self.items.items.len == 0) return &.{};
        const result = try self.allocator.dupe([]u8, self.items.items);
        self.items.clearRetainingCapacity();
        return result;
    }
```

```zig
pub const RtcConfig = struct {

pub const RtcPeer = struct {
    pc: c_int = -1, // peer connection handle
    dc: c_int = -1, // data channel handle
    queue: *MessageQueue,
    allocator: std.mem.Allocator,
    state_queue: *MessageQueue, // for state change notifications

    pub fn init(allocator: std.mem.Allocator, config: RtcConfig, queue: *MessageQueue, state_queue: *MessageQueue) !RtcPeer {
        var rtc_config: c.rtcConfiguration = std.mem.zeroes(c.rtcConfiguration);

        // Build ICE servers array
        var ice_servers: [2][*c]const u8 = undefined;
        var ice_count: c_int = 0;

        const stun_z = try allocator.dupeZ(u8, config.stun_server);
        defer allocator.free(stun_z);
        ice_servers[0] = stun_z.ptr;
        ice_count = 1;

        var turn_z: ?[]u8 = null;
        defer if (turn_z) |tz| allocator.free(tz);

        if (config.turn_server) |turn| {
            turn_z = try allocator.dupeZ(u8, turn);
            ice_servers[1] = turn_z.?.ptr;
            ice_count = 2;
        }

        rtc_config.iceServers = &ice_servers;
        rtc_config.iceServersCount = ice_count;

        const pc = c.rtcCreatePeerConnection(&rtc_config);
        if (pc < 0) return error.CreatePeerConnectionFailed;

        var peer = RtcPeer{
            .pc = pc,
            .queue = queue,
            .state_queue = state_queue,
            .allocator = allocator,
        };

        // Set callbacks — use peer pointer as user data
        const ptr: *RtcPeer = &peer;
        c.rtcSetUserPointer(pc, ptr);

        c.rtcSetLocalDescriptionCallback(pc, onLocalDescription);
        c.rtcSetLocalCandidateCallback(pc, onLocalCandidate);
        c.rtcSetStateChangeCallback(pc, onStateChange);

        return peer;
    }

    pub fn deinit(self: *RtcPeer) void {
        if (self.dc >= 0) c.rtcDeleteDataChannel(self.dc);
        if (self.pc >= 0) c.rtcDeletePeerConnection(self.pc);
        self.dc = -1;
        self.pc = -1;
    }

    /// Set remote SDP description (offer or answer)
    pub fn setRemoteDescription(self: *RtcPeer, sdp: []const u8, sdp_type: []const u8) !void {
        const sdp_z = try self.allocator.dupeZ(u8, sdp);
        defer self.allocator.free(sdp_z);
        const type_z = try self.allocator.dupeZ(u8, sdp_type);
        defer self.allocator.free(type_z);

        if (c.rtcSetRemoteDescription(self.pc, sdp_z.ptr, type_z.ptr) < 0) {
            return error.SetRemoteDescriptionFailed;
        }
    }

    pub fn addRemoteCandidate(self: *RtcPeer, candidate: []const u8, mid: []const u8) !void {
        const cand_z = try self.allocator.dupeZ(u8, candidate);
        defer self.allocator.free(cand_z);
        const mid_z = try self.allocator.dupeZ(u8, mid);
        defer self.allocator.free(mid_z);

        if (c.rtcAddRemoteCandidate(self.pc, cand_z.ptr, mid_z.ptr) < 0) {
            return error.AddCandidateFailed;
        }

    pub fn acceptDataChannel(self: *RtcPeer, dc_handle: c_int) void {

        c.rtcSetUserPointer(dc_handle, @ptrCast(self));

        }

        self.state_queue.push(msg) catch {};
    }

        const mid_str = std.mem.span(mid);
        const msg = std.fmt.allocPrint(self.allocator, "{{\"type\":\"local_candidate\",\"candidate\":\"{s}\",\"mid\":\"{s}\"}}", .{ cand_str, mid_str }) catch return;
        defer self.allocator.free(msg);
        self.state_queue.push(msg) catch {};
    }

    fn onStateChange(pc: c_int, state: c.rtcState, user: ?*anyopaque) callconv(.C) void {
        _ = pc;
        const self: *RtcPeer = @ptrCast(@alignCast(user orelse return));

            c.RTC_NEW => "new",

    }
```

```zig
const std = @import("std");
const posix = std.posix;
const MessageQueue = @import("message_queue.zig").MessageQueue;

/// Minimal WebSocket client for connecting to the signaling server.
/// Handles: connect, send text frames, receive text frames, ping/pong.
pub const SignalClient = struct {
    stream: std.net.Stream,
    allocator: std.mem.Allocator,
    queue: *MessageQueue, // incoming signal messages
    connected: bool = false,
    pairing_code: []const u8,

    /// Connect to the signaling server and perform WebSocket handshake.
    pub fn connect(allocator: std.mem.Allocator, host: []const u8, port: u16, path: []const u8, queue: *MessageQueue, pairing_code: []const u8) !SignalClient {
        const stream = try std.net.tcpConnectToHost(allocator, host, port);

        // WebSocket upgrade request
        var buf: [4096]u8 = undefined;
        const key = "dGhlIHNhbXBsZSBub25jZQ=="; // static for simplicity; production should randomize
        const req = try std.fmt.bufPrint(&buf,
            "GET {s} HTTP/1.1\r\n" ++
            "Host: {s}:{d}\r\n" ++
            "Upgrade: websocket\r\n" ++
            "Connection: Upgrade\r\n" ++
            "Sec-WebSocket-Key: {s}\r\n" ++
            "Sec-WebSocket-Version: 13\r\n" ++
            "\r\n",
            .{ path, host, port, key },
        );

        _ = try stream.write(req);

        // Read response — just verify 101
        var response_buf: [1024]u8 = undefined;
        const n = try stream.read(&response_buf);
        if (n == 0) return error.ConnectionClosed;
        const response = response_buf[0..n];
        if (std.mem.indexOf(u8, response, "101") == null) return error.WebSocketUpgradeFailed;

        var client = SignalClient{
            .stream = stream,
            .allocator = allocator,
            .queue = queue,
            .connected = true,
            .pairing_code = try allocator.dupe(u8, pairing_code),
        };

        return client;
    }

    pub fn deinit(self: *SignalClient) void {
        self.stream.close();
        self.allocator.free(self.pairing_code);
        self.connected = false;
    }

    /// Send a text WebSocket frame (masked, as required for clients).
    pub fn sendText(self: *SignalClient, payload: []const u8) !void {
        // WebSocket client frame: FIN=1, opcode=1 (text), MASK=1
        var header: [14]u8 = undefined;
        var header_len: usize = 2;
        header[0] = 0x81; // FIN + text

        if (payload.len < 126) {
            header[1] = @as(u8, @intCast(payload.len)) | 0x80; // MASK bit
        } else if (payload.len < 65536) {
            header[1] = 126 | 0x80;
            header[2] = @intCast(payload.len >> 8);
            header[3] = @intCast(payload.len & 0xFF);
            header_len = 4;
        } else {
            return error.PayloadTooLarge;
        }

        // Masking key (4 bytes)
        var mask: [4]u8 = undefined;
        std.crypto.random.bytes(&mask);
        @memcpy(header[header_len .. header_len + 4], &mask);
        header_len += 4;

        _ = try self.stream.write(header[0..header_len]);

        // Send masked payload
        var masked = try self.allocator.alloc(u8, payload.len);
        defer self.allocator.free(masked);
        for (payload, 0..) |byte, i| {
            masked[i] = byte ^ mask[i % 4];
        }
        _ = try self.stream.write(masked);
    }

    /// Send a JSON message to the signaling server.
    pub fn sendJson(self: *SignalClient, json: []const u8) !void {
        try self.sendText(json);
    }

    /// Register as daemon with the signaling server.
    pub fn register(self: *SignalClient) !void {
        const msg = try std.fmt.allocPrint(self.allocator,
            "{{\"type\":\"register\",\"pairing_code\":\"{s}\"}}",
            .{self.pairing_code},
        );
        defer self.allocator.free(msg);
        try self.sendText(msg);
    }

    /// Read loop — runs in its own thread, pushes messages to queue.
    pub fn readLoop(self: *SignalClient) void {
        var buf: [65536]u8 = undefined;
        while (self.connected) {
            // Read WebSocket frame header
            var header: [2]u8 = undefined;
            _ = self.stream.read(&header) catch break;

            const opcode = header[0] & 0x0F;
            const masked = (header[1] & 0x80) != 0;
            var payload_len: usize = header[1] & 0x7F;

            if (payload_len == 126) {
                var ext: [2]u8 = undefined;
                _ = self.stream.read(&ext) catch break;
                payload_len = (@as(usize, ext[0]) << 8) | ext[1];
            }

            // Skip mask key if present (server shouldn't mask, but handle it)
            var mask_key: [4]u8 = undefined;
            if (masked) {
                _ = self.stream.read(&mask_key) catch break;
            }

            if (payload_len > buf.len) break; // too large

            // Read payload
            var total: usize = 0;
            while (total < payload_len) {
                const n = self.stream.read(buf[total..payload_len]) catch break;
                if (n == 0) break;
                total += n;
            }

            if (masked) {
                for (buf[0..total], 0..) |*byte, i| {
                    byte.* ^= mask_key[i % 4];
                }
            }

            switch (opcode) {
                0x1 => { // Text frame
                    self.queue.push(buf[0..total]) catch {};
                },
                0x9 => { // Ping — respond with pong
                    var pong_header: [2]u8 = .{ 0x8A, 0x80 }; // FIN + pong + MASK
                    var pong_mask: [4]u8 = undefined;
                    std.crypto.random.bytes(&pong_mask);
                    _ = self.stream.write(&pong_header) catch {};
                    _ = self.stream.write(&pong_mask) catch {};
                },
                0x8 => { // Close
                    break;
                },
                else => {},
            }
        }
        self.connected = false;
    }
};
```

```zig
fn handleDataChannelMessage(
    allocator: std.mem.Allocator,
    raw: []const u8,
    session_manager: *SessionManager,
    auth_ptr: *Auth,
    rtc_peer: *?*RtcPeer,
) void {
    const parsed = protocol.parseClientMessage(allocator, raw) catch return;
    defer parsed.deinit();
    const msg = parsed.value();
    const msg_type = msg.@"type";

    if (std.mem.eql(u8, msg_type, "auth")) {
        // Handle setup_secret or session_token auth via DataChannel
        if (msg.token) |token| {
            // Try as setup_secret first
            if (auth_ptr.validateSetupToken(token)) |session_token| {
                const result = protocol.encodeAuthResult(allocator, true, &session_token) catch return;
                defer allocator.free(result);
                if (rtc_peer.*) |peer| peer.send(result) catch {};
                // Send sessions_sync after auth
                sendSessionsSync(allocator, session_manager, rtc_peer);
            } else if (auth_ptr.validateSessionToken(token)) {
                const result = protocol.encodeAuthResult(allocator, true, null) catch return;
                defer allocator.free(result);
                if (rtc_peer.*) |peer| peer.send(result) catch {};
                sendSessionsSync(allocator, session_manager, rtc_peer);
            } else {
                const result = protocol.encodeAuthResult(allocator, false, null) catch return;
                defer allocator.free(result);
                if (rtc_peer.*) |peer| peer.send(result) catch {};
            }
        }
    } else if (std.mem.eql(u8, msg_type, "terminal_input")) {
        if (msg.session_id) |sid| {
            if (msg.data) |data| session_manager.writeToSession(sid, data);
        }
    } else if (std.mem.eql(u8, msg_type, "resize")) {
        if (msg.session_id) |sid| {
            session_manager.resizeSession(sid, msg.rows orelse 24, msg.cols orelse 80);
        }
    } else if (std.mem.eql(u8, msg_type, "prompt_response")) {
        if (msg.session_id) |sid| {
            if (msg.text) |text| session_manager.resolvePromptResponse(sid, text);
        }
    } else if (std.mem.eql(u8, msg_type, "create_session")) {
        const command = msg.data orelse "claude";
        const sid = session_manager.createSession(.{ .command = command }) catch return;
        const result = protocol.encodeCreateSessionResult(allocator, sid) catch return;
        defer allocator.free(result);
        if (rtc_peer.*) |peer| peer.send(result) catch {};
    } else if (std.mem.eql(u8, msg_type, "delete_session")) {
        if (msg.session_id) |sid| {
            session_manager.destroySession(sid);
            const result = protocol.encodeDeleteSessionResult(allocator, sid, true) catch return;
            defer allocator.free(result);
            if (rtc_peer.*) |peer| peer.send(result) catch {};
        }
    } else if (std.mem.eql(u8, msg_type, "ping")) {
        if (rtc_peer.*) |peer| peer.send(protocol.encodePong()) catch {};
    }
}

fn sendSessionsSync(allocator: std.mem.Allocator, session_manager: *SessionManager, rtc_peer: *?*RtcPeer) void {
    const sessions = session_manager.listSessions(allocator) catch return;
    defer {
        for (sessions) |s| SessionManager.freeSessionInfo(allocator, s);
        allocator.free(sessions);
    }

    // Build sessions JSON array
    var json_buf = std.ArrayList(u8).empty;
    defer json_buf.deinit(allocator);
    json_buf.append(allocator, '[') catch return;
    for (sessions, 0..) |s, i| {
        if (i > 0) json_buf.append(allocator, ',') catch return;
        const entry = std.fmt.allocPrint(allocator,
            "{{\"id\":{d},\"state\":\"{s}\",\"command\":\"{s}\",\"cwd\":\"{s}\"}}",
            .{ s.id, @tagName(s.state), s.command, s.cwd },
        ) catch return;
        defer allocator.free(entry);
        json_buf.appendSlice(allocator, entry) catch return;
    }
    json_buf.append(allocator, ']') catch return;

    const msg = protocol.encodeSessionsSyncJson(allocator, json_buf.items) catch return;
    defer allocator.free(msg);
    if (rtc_peer.*) |peer| peer.send(msg) catch {};
}
```

```typescript
export class SignalClient {
  private ws: WebSocket | null = null;
  private handlers: MessageHandler[] = [];
  private reconnectTimer: number | null = null;
  private url: string;
  private pairingCode: string;

  constructor(url: string, pairingCode: string) {
    this.url = url;
    this.pairingCode = pairingCode;
  }

  connect(): Promise<void> {
    return new Promise((resolve, reject) => {
      this.ws = new WebSocket(this.url);
      this.ws.onopen = () => {
        // Join room with pairing code
        this.send({ type: 'join', pairing_code: this.pairingCode });
        resolve();
      };
      this.ws.onmessage = (ev) => {
        try {
          const msg: SignalMessage = JSON.parse(ev.data);
          this.handlers.forEach((h) => h(msg));
        } catch {}
      };
      this.ws.onclose = () => this.scheduleReconnect();
      this.ws.onerror = () => {
        reject(new Error('WebSocket connection failed'));
        this.ws?.close();
      };
    });
  }

  private scheduleReconnect() {
    if (this.reconnectTimer) return;
    this.reconnectTimer = window.setTimeout(() => {
      this.reconnectTimer = null;
      this.connect().catch(() => {});
    }, 2000);
  }

  onMessage(handler: MessageHandler): () => void {
    this.handlers.push(handler);
    return () => { this.handlers = this.handlers.filter((h) => h !== handler); };
  }

  sendSdpOffer(sdp: string, sdpType: string) {
    this.send({ type: 'sdp_offer', sdp, sdp_type: sdpType });
  }

  sendIceCandidate(candidate: string, mid: string) {
    this.send({ type: 'ice_candidate', candidate, mid });
  }

  private send(msg: Record<string, unknown>) {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(msg));
    }
  }

  disconnect() {
    if (this.reconnectTimer) { clearTimeout(this.reconnectTimer); this.reconnectTimer = null; }
    this.ws?.close();
    this.ws = null;
  }
}
```

```typescript
import { SignalClient } from './signal';

type MessageHandler = (msg: ServerMessage) => void;

export class RtcManager {
  private pc: RTCPeerConnection | null = null;
  private dc: RTCDataChannel | null = null;
  private signal: SignalClient | null = null;
  private handlers: MessageHandler[] = [];
  private authenticated = false;
  private pingInterval: number | null = null;

  async connect(signalUrl: string, pairingCode: string, stunServer = 'stun:stun.l.google.com:19302'): Promise<void> {
    // Connect to signaling server
    this.signal = new SignalClient(signalUrl, pairingCode);

    this.signal.onMessage((msg) => {
      if (msg.type === 'peer_joined') {
        // Daemon is ready, we are the browser — we create the offer
        this.startWebRTC(stunServer);
      } else if (msg.type === 'sdp_answer') {
        this.handleSdpAnswer(msg.sdp as string, msg.sdp_type as string);
      } else if (msg.type === 'ice_candidate') {
        this.handleRemoteCandidate(msg.candidate as string, msg.mid as string);
      } else if (msg.type === 'peer_left') {
        this.handlePeerLeft();
      } else if (msg.type === 'error') {
        console.error('[Signal] error:', msg.message);
      }
    });

    await this.signal.connect();

    // Wait for peer_joined — daemon may already be registered, signal server sends it on join
  }

  private async startWebRTC(stunServer: string) {
    this.pc = new RTCPeerConnection({
      iceServers: [{ urls: stunServer }],
    });

    // Create DataChannel before creating offer
    this.dc = this.pc.createDataChannel('kite', { ordered: true });
    this.setupDataChannel(this.dc);

    this.pc.onicecandidate = (ev) => {
      if (ev.candidate && this.signal) {
        this.signal.sendIceCandidate(ev.candidate.candidate, ev.candidate.sdpMid ?? '');
      }
    };

    this.pc.onconnectionstatechange = () => {
      console.log('[RTC] connection state:', this.pc?.connectionState);
      if (this.pc?.connectionState === 'disconnected' || this.pc?.connectionState === 'failed') {
        this.handlePeerLeft();
      }
    };

    // Create and send offer
    const offer = await this.pc.createOffer();
    await this.pc.setLocalDescription(offer);
    this.signal?.sendSdpOffer(offer.sdp!, offer.type);
  }

  private setupDataChannel(dc: RTCDataChannel) {
    dc.onopen = () => {
      console.log('[RTC] DataChannel open');
      this.startPing();
    };

    dc.onmessage = (ev) => {
      try {
        const msg: ServerMessage = JSON.parse(ev.data);
        if (msg.type === 'pong') return; // heartbeat response
        this.handlers.forEach((h) => h(msg));
      } catch {}
    };

    dc.onclose = () => {
      console.log('[RTC] DataChannel closed');
      this.stopPing();
    };
  }

  private async handleSdpAnswer(sdp: string, sdpType: string) {
    if (!this.pc) return;
    await this.pc.setRemoteDescription(new RTCSessionDescription({
      sdp,
      type: sdpType as RTCSdpType,
    }));
  }

  private async handleRemoteCandidate(candidate: string, mid: string) {
    if (!this.pc) return;
    await this.pc.addIceCandidate(new RTCIceCandidate({
      candidate,
      sdpMid: mid,
    }));
  }

  private handlePeerLeft() {
    this.stopPing();
    this.dc?.close();
    this.pc?.close();
    this.dc = null;
    this.pc = null;
    this.authenticated = false;
    // Notify handlers of disconnection
    this.handlers.forEach((h) => h({ type: 'disconnected' } as any));
  }

  private startPing() {
    this.stopPing();
    this.pingInterval = window.setInterval(() => {
      this.sendRaw({ type: 'ping' });
    }, 10_000);
  }

  private stopPing() {
    if (this.pingInterval) { clearInterval(this.pingInterval); this.pingInterval = null; }
  }

  // Public API — matches WsManager interface

  onMessage(handler: MessageHandler): () => void {
    this.handlers.push(handler);
    return () => { this.handlers = this.handlers.filter((h) => h !== handler); };
  }

  /** Send auth via DataChannel (setup_secret or session_token) */
  authenticate(token: string) {
    this.sendRaw({ type: 'auth', token });
  }

  sendTerminalInput(data: string, sessionId: number) {
    this.sendRaw({ type: 'terminal_input', data, session_id: sessionId });
  }

  sendResize(cols: number, rows: number, sessionId: number) {
    this.sendRaw({ type: 'resize', cols, rows, session_id: sessionId });
  }

  sendPromptResponse(text: string, sessionId: number) {
    this.sendRaw({ type: 'prompt_response', text, session_id: sessionId });
  }

  createSession(command = 'claude') {
    this.sendRaw({ type: 'create_session', data: command });
  }

  deleteSession(sessionId: number) {
    this.sendRaw({ type: 'delete_session', session_id: sessionId });
  }

  isOpen(): boolean {
    return this.dc?.readyState === 'open';
  }

  private sendRaw(msg: Record<string, unknown>) {
    if (this.dc?.readyState === 'open') {
      this.dc.send(JSON.stringify(msg));
    } else {
      console.warn('[RTC] send FAILED - DataChannel not open, state:', this.dc?.readyState);
    }
  }

  disconnect() {
    this.stopPing();
    this.signal?.disconnect();
    this.dc?.close();
    this.pc?.close();
    this.signal = null;
    this.dc = null;
    this.pc = null;
  }
}
```

```typescript
  }

  sorted(): SessionInfo[] {
    const priority: Record<string, number> = { asking: 0, waiting: 0, waiting_permission: 0, running: 1 };
    return [...this.sessions]
      .filter((s) => s.state !== 'stopped')
      .sort((a, b) => (priority[a.state] ?? 9) - (priority[b.state] ?? 9));
  }
}

export const sessionStore = new SessionStore();
rtc.onMessage((msg) => sessionStore.handleMessage(msg));
```

```svelte
<script lang="ts">
  import SessionList from './components/SessionList.svelte';
  import SessionDetail from './components/SessionDetail.svelte';
  import { rtc } from './lib/webrtc';
  import { onMount } from 'svelte';
  import {
    clearStoredToken, clearStoredPairingCode, clearPairingFromHash,
    getStoredToken, getStoredPairingCode, setStoredToken, setStoredPairingCode,
    parsePairingFromHash,
  } from './lib/auth';

  let currentView = $state<'list' | 'detail'>('list');
  let selectedSessionId = $state<number | null>(null);
  let authReady = $state(false);
  let authRequired = $state(false);
  let authError = $state('');
  let pairingInput = $state('');
  let connecting = $state(false);

  // Signal server URL — same origin as the page
  const signalUrl = `${location.protocol === 'https:' ? 'wss:' : 'ws:'}//${location.host}/ws`;

  onMount(() => {
    const unsubscribe = rtc.onMessage((msg) => {
      if (msg.type === 'auth_result') {
        if ((msg as any).success) {
          if ((msg as any).token) {
            setStoredToken((msg as any).token);
          }
          authReady = true;
          authRequired = false;
          authError = '';
          connecting = false;
        } else {
          clearStoredToken();
          clearStoredPairingCode();
          authReady = false;
          authRequired = true;
          authError = 'Authentication failed.';
          connecting = false;
        }
      }
    });

    void initializeAuth();

    return () => {
      unsubscribe();
      rtc.disconnect();
    };
  });

  async function initializeAuth() {
    // Check URL hash for pairing info: #/pair/{code}:{secret}
    const pairing = parsePairingFromHash();
    if (pairing) {
      clearPairingFromHash();
      await connectAndAuth(pairing.pairingCode, pairing.setupSecret);
      return;
    }

    // Check localStorage for existing session
    const storedToken = getStoredToken();
    const storedCode = getStoredPairingCode();
    if (storedToken && storedCode) {
      await connectAndAuth(storedCode, storedToken);
      return;
    }

    // No credentials — show pairing input
    authRequired = true;
  }

  async function connectAndAuth(pairingCode: string, secret: string) {
    connecting = true;
    authError = '';
    try {
      await rtc.connect(signalUrl, pairingCode);
      setStoredPairingCode(pairingCode);

      // Wait for DataChannel to open, then authenticate
      const waitForOpen = () => new Promise<void>((resolve, reject) => {
        const check = setInterval(() => {
          if (rtc.isOpen()) { clearInterval(check); resolve(); }
        }, 100);
        setTimeout(() => { clearInterval(check); reject(new Error('timeout')); }, 15000);
      });

      await waitForOpen();
      rtc.authenticate(secret);
    } catch {
      connecting = false;
      clearStoredToken();
      clearStoredPairingCode();
      authRequired = true;
      authError = 'Connection failed. Check pairing code and try again.';
    }
  }

  async function submitPairing() {
    authError = '';
    const trimmed = pairingInput.trim();
    if (!trimmed) return;

    // Parse "code:secret" format
    const parts = trimmed.split(':');
    if (parts.length !== 2) {
      authError = 'Invalid format. Expected: pairing_code:setup_secret';
      return;
    }

    await connectAndAuth(parts[0], parts[1]);
  }

  function handleAuthKeydown(event: KeyboardEvent) {
    if (event.key === 'Enter') {
      event.preventDefault();
      void submitPairing();
    }
  }

  function openSession(id: number) { selectedSessionId = id; currentView = 'detail'; }
  function goBack() { currentView = 'list'; selectedSessionId = null; }
</script>
```

```zig
    defer allocator.free(config_json);

    defer file.close();
    try file.writeAll(config_json);
```
