// ── Fuzzy matching ──

/// Compute Levenshtein edit distance between two strings.
/// Uses a single-row buffer to minimize memory.
pub fn levenshteinDistance(a: []const u8, b: []const u8) usize {
    if (a.len == 0) return b.len;
    if (b.len == 0) return a.len;

    // Use stack buffer for small strings, fallback for large
    var buf: [256]usize = undefined;
    if (b.len + 1 > buf.len) return a.len; // too long, no match
    const row = buf[0 .. b.len + 1];

    for (row, 0..) |*v, i| v.* = i;

    for (a, 0..) |ca, i| {
        var prev = i;
        row[0] = i + 1;
        for (b, 0..) |cb, j| {
            const cost: usize = if (ca == cb) 0 else 1;
            const ins = row[j + 1] + 1;
            const del = row[j] + 1;
            const sub = prev + cost;
            prev = row[j + 1];
            row[j + 1] = @min(ins, @min(del, sub));
        }
    }
    return row[b.len];
}

/// Find the closest match from candidates within the given threshold.
/// Returns null if no match is close enough.
pub fn findClosest(input: []const u8, candidates: []const []const u8, threshold: usize) ?[]const u8 {
    var best: ?[]const u8 = null;
    var best_dist: usize = threshold + 1;

    for (candidates) |c| {
        const d = levenshteinDistance(input, c);
        if (d < best_dist) {
            best_dist = d;
            best = c;
        }
    }
    return best;
}
```

// ── Error messages ──

pub const command_names = [_][]const u8{ "start", "run", "hook", "setup", "status", "help" };

pub fn printUnknownCommand(input: []const u8) void {
    const stderr_file = std.fs.File.stderr();
    var buf: [1024]u8 = undefined;
    var w = stderr_file.writer(&buf);
    const out = &w.interface;

    out.print("Unknown command: '{s}'.", .{input}) catch {};
    if (findClosest(input, &command_names, 2)) |suggestion| {
        out.print(" Did you mean '{s}{s}{s}'?", .{ code(.green), suggestion, code(.reset) }) catch {};
    }
    out.print("\n\nRun '{s}kite help{s}' for a list of commands.\n", .{ code(.cyan), code(.reset) }) catch {};
    out.flush() catch {};
}

pub fn printUnknownOption(option: []const u8, known_options: []const []const u8, command: []const u8) void {
    const stderr_file = std.fs.File.stderr();
    var buf: [1024]u8 = undefined;
    var w = stderr_file.writer(&buf);
    const out = &w.interface;

    out.print("Unknown option: '{s}'.", .{option}) catch {};
    if (findClosest(option, known_options, 2)) |suggestion| {
        out.print(" Did you mean '{s}{s}{s}'?", .{ code(.yellow), suggestion, code(.reset) }) catch {};
    }
    out.print("\n\nRun '{s}kite {s} --help{s}' for usage.\n", .{ code(.cyan), command, code(.reset) }) catch {};
    out.flush() catch {};
}

pub fn printMissingOption(option: []const u8, command: []const u8, usage: []const u8) void {
    const stderr_file = std.fs.File.stderr();
    var buf: [1024]u8 = undefined;
    var w = stderr_file.writer(&buf);
    const out = &w.interface;

    out.print("Missing required option: {s}{s}{s}\n\n{s}Usage:{s}\n  {s}\n", .{
        code(.yellow), option, code(.reset),
        code(.bold),   code(.reset),
        usage,
    }) catch {};
    _ = command;
    out.flush() catch {};
}
```

### Task 6: Create GitHub Actions Release Workflow

**Files:**
- Create: `.github/workflows/release.yml`

- [ ] **Step 1: Create the workflow file**

```yaml
name: Release

on:
  push:
    tags:
      - 'v*'

permissions:
  contents: write

jobs:
  build:
    strategy:
      matrix:
        include:
          - os: macos-14
            arch: arm64
            artifact: kite-darwin-arm64
          - os: macos-13
            arch: amd64
            artifact: kite-darwin-amd64
          - os: ubuntu-24.04
            arch: amd64
            artifact: kite-linux-amd64
          - os: ubuntu-24.04
            arch: arm64
            artifact: kite-linux-arm64
            cross: true

    runs-on: ${{ matrix.os }}

    steps:
      - uses: actions/checkout@v4
        with:
          submodules: recursive

      - name: Install Zig
        uses: mlugg/setup-zig@v2
        with:
          version: 0.15.2

      - name: Install Node.js
        uses: actions/setup-node@v4
        with:
          node-version: 20

      - name: Build web UI
        run: cd web && npm ci && npm run build

      # macOS: install libdatachannel via Homebrew
      - name: Install libdatachannel (macOS)
        if: runner.os == 'macOS'
        run: brew install libdatachannel

      # Linux: build libdatachannel from source (static)
      - name: Build libdatachannel (Linux)
        if: runner.os == 'Linux'
        run: |
          sudo apt-get update
          sudo apt-get install -y cmake g++ libssl-dev
          git clone --depth 1 --recurse-submodules https://github.com/nichos1983/libdatachannel.git /tmp/libdatachannel
          # Note: for arm64 cross-compile, libdatachannel must also be cross-compiled.
          # This may require a separate cmake invocation with -DCMAKE_SYSTEM_PROCESSOR=aarch64.
          cd /tmp/libdatachannel
          cmake -B build -DBUILD_SHARED_LIBS=OFF -DCMAKE_INSTALL_PREFIX=/usr/local
          cmake --build build -j$(nproc)
          sudo cmake --install build

      # Native build (non-cross)
      - name: Build kite (native)
        if: ${{ !matrix.cross }}
        run: |
          LIB_PATH=${{ runner.os == 'macOS' && '/opt/homebrew' || '/usr/local' }}
          zig build -Doptimize=ReleaseSafe -Dlibdatachannel-path=$LIB_PATH

      # Cross-compile for Linux arm64
      - name: Build kite (cross-compile arm64)
        if: ${{ matrix.cross }}
        run: |
          zig build -Doptimize=ReleaseSafe -Dtarget=aarch64-linux-gnu -Dlibdatachannel-path=/usr/local

      - name: Package
        run: |
          cd zig-out/bin
          tar czf ${{ matrix.artifact }}.tar.gz kite

      - name: Upload artifact
        uses: actions/upload-artifact@v4
        with:
          name: ${{ matrix.artifact }}
          path: zig-out/bin/${{ matrix.artifact }}.tar.gz

  release:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - name: Download all artifacts
        uses: actions/download-artifact@v4
        with:
          path: artifacts

      - name: Create GitHub Release
        uses: softprops/action-gh-release@v2
        with:
          files: artifacts/**/*.tar.gz
          generate_release_notes: true
```
